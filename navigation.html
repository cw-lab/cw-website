$(function(){
  $("nav.sidebar > ul > li > i.more").click(function(event) {
    $(this).parent().siblings().children("i.more").removeClass("active");
    $(this).parent().siblings().children("ul").slideUp();
    $(this).siblings("ul").slideToggle();
    $(this).toggleClass("active");
  });
});
